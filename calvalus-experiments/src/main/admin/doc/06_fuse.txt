# on each node do
# first become root
sudo su -

# install hadoop-fuse-dfs software
apt-get update
apt-get -y install hadoop-0.20-fuse

# create mount point
mkdir /mnt/hdfs

# append line to fstab
cp -p /etc/fstab /etc/fstab.orig
echo '# hadoop fuse' >> /etc/fstab
echo 'hadoop-fuse-dfs#dfs://cvmaster00:9000   /mnt/hdfs       fuse    big_writes,notrash,rw   0       0' >> /etc/fstab

# mount hdfs by fuse
mount /mnt/hdfs

exit
exit
