
#### Commands, first become root using
sudo su -

parted /dev/sdb --script -- mklabel msdos mkpart primary 0 -1
parted /dev/sdc --script -- mklabel msdos mkpart primary 0 -1

mkfs.ext4 /dev/sdb1
mkfs.ext4 /dev/sdc1

# create mount points
mkdir /hd1
mkdir /hd2


grep UUID /etc/fstab

# first the UUID is stored in a environment variable
# then a line is appened to fstab using this environment variable
# the 'echo' command is embedded in a 'sh' call because otherwise the append command would not be subject to the 'sudo'

UUID=`blkid -o value -s UUID /dev/sdb1` && echo "UUID=${UUID} /hd1            ext4    noatime,errors=remount-ro    0     2" >> /etc/fstab
UUID=`blkid -o value -s UUID /dev/sdc1` && echo "UUID=${UUID} /hd2            ext4    noatime,errors=remount-ro    0     2" >> /etc/fstab

#### mount it now
mount /hd1
mount /hd2

# create hadoop directories on the new disks
mkdir /hd1/hadoop
chown hadoop:hadoop /hd1/hadoop
chmod 777 /hd1/hadoop

mkdir /hd2/hadoop
chown hadoop:hadoop /hd2/hadoop
chmod 777 /hd2/hadoop

# all in one paste line
mkdir /hd1/hadoop && chown hadoop:hadoop /hd1/hadoop && chmod 777 /hd1/hadoop && mkdir /hd2/hadoop && chown hadoop:hadoop /hd2/hadoop && chmod 777 /hd2/hadoop
