# add cloudera deb package repository
echo "deb http://archive.cloudera.com/debian lucid-cdh3 contrib" > /etc/apt/sources.list.d/cloudera.list

# add cloudera key to keyring
wget -O- http://archive.cloudera.com/debian/archive.key | apt-key add -

# install
apt-get update
apt-get -y install hadoop-0.20 hadoop-0.20-pipes hadoop-0.20-tasktracker hadoop-0.20-datanode

# on the master node only (no datanode here)
apt-get -y install hadoop-0.20 hadoop-0.20-pipes hadoop-0.20-jobtracker hadoop-0.20-namenode hadoop-0.20-secondarynamenode
